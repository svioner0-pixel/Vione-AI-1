import { NextRequest } from 'next/server';

export async function POST(request: NextRequest) {
  try {
    const formData = await request.formData();
    const userImage = formData.get('image') as File;

    if (!userImage) {
      return new Response('Görsel eksik', { status: 400 });
    }

    const bytes = await userImage.arrayBuffer();
    const buffer = Buffer.from(bytes);
    const base64Image = `data:${userImage.type};base64,${buffer.toString('base64')}`;

    // En stabil undress yaklaşımı: Basit efekt + güçlü prompt
    const eternalRes = await fetch('https://open.eternalai.org/generate', {
      method: 'POST',
      headers: {
        'x-api-key': process.env.ETERNAL_AI_API_KEY!,
        'Content-Type': 'application/json',
        'accept': 'application/json',
      },
      body: JSON.stringify({
        images: [base64Image],
        effect_id: "1",                     // En basit ve stabil efekt (docs'ta önerilen)
        prompt: "completely remove all clothes, fully nude, naked body, realistic skin texture, detailed anatomy, no clothing",
      }),
    });

    if (!eternalRes.ok) {
      const errorText = await eternalRes.text();
      console.error('Eternal AI Raw Error:', errorText);
      return new Response(`Eternal AI hatası: ${eternalRes.status}`, { status: 500 });
    }

    const data = await eternalRes.json();
    let imageUrl = data.image_url || data.output || '';

    // Eğer request_id varsa sonucu bekle
    if (data.request_id && !imageUrl) {
      console.log('Polling başladı...');
      await new Promise(r => setTimeout(r, 8000)); // 8 saniye bekle (undress biraz uzun sürebilir)

      const pollRes = await fetch(`https://open.eternalai.org/poll-result/${data.request_id}`, {
        headers: { 'x-api-key': process.env.ETERNAL_AI_API_KEY! }
      });

      const pollData = await pollRes.json();
      imageUrl = pollData.image_url || pollData.output || '';
    }

    if (!imageUrl) {
      return new Response('Sonuç resmi alınamadı', { status: 500 });
    }

    // Sonucu kullanıcıya dön
    const finalRes = await fetch(imageUrl);
    const imageBuffer = await finalRes.arrayBuffer();

    return new Response(imageBuffer, {
      headers: { 'Content-Type': 'image/png' },
    });

  } catch (error) {
    console.error('Genel hata:', error);
    return new Response('Sunucu hatası', { status: 500 });
  }
}
